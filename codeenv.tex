% java (kind of "Eclipse" highlighting)

	language=JAVA,%
	commentstyle=\color{comment},%
	stringstyle=\color{string},%

	numbers=left,
	numberstyle=\tiny\color{black},
	stepnumber=1,
	showstringspaces=false,%
	columns=fixed,%
	moredelim=[is][\itshape]{@@}{@@},%
	tabsize=4
}
